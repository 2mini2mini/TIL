## 📖 2차원 배열

-> 2차원 배열은 배열의 각 요소가 배열인 경우를 말한다
EX) const arr = [
[1, 2], // 0번째 행
[3, 4], // 1번째 행
[5, 6], // 2번째 행
[7, 8], // 3번째 행
];

### ✏️ 2차원 배열 생성

        <script>
        // 4행 2열 2차원 배열 생성
        const arr = [
                [1, 2], //0번째 행
                [3, 4], //1번째 행
                [5, 6], //2번째 행
                [7, 8], //3번째 행
        ];
        let text = "";
        text += arr[2][0] + "<br>"; //2번째 행의 1열
        text += arr[2][1]; //2번째 행의 2열
        document.getElementById("show").innerHTML = text;
        </script>
